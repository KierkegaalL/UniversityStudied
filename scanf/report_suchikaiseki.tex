#include<math.h>
#define h 0.5


double dy(double x, double y);

    double x, x_max, y;

    {
        printf("%lf \n",y);
        x  = x + h;
        y  = y + dy(x,y) * h;
    }
    return 0;
}

}

\begin{breakitembox}[l]{ルンゲ・クッタ法} \small
\begin{verbatim}

/********************************************
名前： RungeKutta.c
機能： ルンゲ・クッタ法を用いて微分方程式の解を求める
作者： 15024031 奥　龍司
特記事項：特になし
********************************************/
#include<stdio.h>
#include<math.h>
#define h 0.5

double dy(double x, double y);

int main(void){
    double x, x_max, y, k1, k2, k3, k4;
    
    y = 0.5;
    
    x = 0;
    x_max = 10;
    
    while(x < x_max)
    {
        printf("%lf \n",y);
        x  = x + h;
        k1 = h * dy(x, y);
        k2 = h * dy(x + h/2, y + k1/2);
        k3 = h * dy(x + h/2, y + k2/2);
        k4 = h * dy(x + h, y + k3);
        y  = y + (k1 + 2*k2 + 2*k3 + k4)/6;
    }
    
  return 0;
}

double dy(double x, double y){//求める微分方程式
    return 2 * y/(1 + x);
}

\end{verbatim}
\end{breakitembox}
